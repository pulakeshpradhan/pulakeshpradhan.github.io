  max: 0.3
};

Functions in Earth Engine 
// Functions in Earth Engine

// Simple function
function calculateNDVI(image) {
  return image.normalizedDifference(['NIR', 'RED']).rename('NDVI');
}

// Using the function
var landsat = ee.Image('LANDSAT/LC08/C02/T1_TOA/LC08_044034_20140318');
var ndvi = calculateNDVI(landsat);

// Anonymous function with ee.Image.map()
var collection = ee.ImageCollection('LANDSAT/LC08/C02/T1_TOA')
  .filterDate('2020-01-01', '2020-12-31')
  .map(function(image) {
    return image.normalizedDifference(['B5', 'B4']).rename('NDVI');
  });

// Arrow function (ES6)
var addTimeBand = (image) => {
  return image.addBands(image.metadata('system:time_start'));

Importing and Exporting Data 
// Importing and Exporting Data

// Load a public dataset
var s2 = ee.ImageCollection('COPERNICUS/S2_SR')
  .filterDate('2020-06-01', '2020-09-01')
  .filterBounds(ee.Geometry.Point([-122.0841, 37.4223]));

// Create a median composite
var composite = s2.median();

// Display the composite
var vizParams = {bands: ['B4', 'B3', 'B2'], min: 0, max: 3000};
Map.addLayer(composite, vizParams, 'Sentinel-2 Composite');

// Export the image to Google Drive
Export.image.toDrive({
  image: composite,
  description: 'Sentinel2_Composite',
  scale: 10,
  region: composite.geometry().bounds(),
  maxPixels: 1e9
});

// Load a feature collection (countries)
var countries = ee.FeatureCollection('USDOS/LSIB_SIMPLE/2017');

// Filter to a specific country
var country = countries.filter(ee.Filter.eq('country_na', 'United States'));

// Export the feature collection to CSV
Export.table.toDrive({
  collection: country,
  description: 'US_Boundary',
  fileFormat: 'SHP'
});

Day 5: Core Functionality – Filtering and Functions

  • Filtering Data: filterBounds(), filterDate(), filter(.....)
  • Using Map and Reduce functions

Filtering and Functions 
// Filtering and Functions

// Define a region of interest
var roi = ee.Geometry.Point([-122.0841, 37.4223]).buffer(50000);

// Load Sentinel-2 collection
var s2Collection = ee.ImageCollection('COPERNICUS/S2_SR')
  .filterDate('2020-01-01', '2020-12-31')  // Filter by date
  .filterBounds(roi)  // Filter by location
  .filter(ee.Filter.lt('CLOUDY_PIXEL_PERCENTAGE', 20));  // Filter by metadata

print('Collection size after filtering:', s2Collection.size());

// Define a function to calculate NDVI
function addNDVI(image) {
  var ndvi = image.normalizedDifference(['B8', 'B4']).rename('NDVI');
  return image.addBands(ndvi);
}

// Map the function over the collection
var withNDVI = s2Collection.map(addNDVI);

// Select just the NDVI bands
var ndviCollection = withNDVI.select('NDVI');

// Calculate the mean NDVI
var meanNDVI = ndviCollection.mean();

// Display the result
var ndviParams = {min: 0, max: 1, palette: ['red', 'yellow', 'green']};
Map.centerObject(roi, 9);
Map.addLayer(meanNDVI, ndviParams, 'Mean NDVI');
Map.addLayer(roi, {color: 'blue'}, 'Region of Interest');

Day 6: Core Functionality – Reducers

  • Introduction to Reducers: reduceRegion(), reduceColumns(), reduceNeighbors()
  • Zonal Statistics: Mean, Max, Min, Sum using Reducers
  • Reducing ImageCollections to single images
Reduce Region Example 
// Reduce Region Example

// Define a region of interest
var roi = ee.Geometry.Rectangle([-122.5, 37.5, -122.0, 38.0]);

// Load a Landsat image
var image = ee.Image('LANDSAT/LC08/C02/T1_TOA/LC08_044034_20140318');

// Calculate NDVI
var ndvi = image.normalizedDifference(['B5', 'B4']).rename('NDVI');

// Compute statistics for the region
var stats = ndvi.reduceRegion({
  reducer: ee.Reducer.mean().combine({
    reducer2: ee.Reducer.minMax(),
    sharedInputs: true
  }),
  geometry: roi,
  scale: 30,
  maxPixels: 1e9
});

print('NDVI Statistics:', stats);

// Display the region and NDVI
Map.centerObject(roi, 10);
Map.addLayer(ndvi, {min: 0, max: 1, palette: ['red', 'yellow', 'green']}, 'NDVI');
Map.addLayer(roi, {color: 'blue'}, 'ROI');

Reduce Collection Example 
// Reduce Collection Example

// Define a region of interest
var roi = ee.Geometry.Point([-122.0841, 37.4223]).buffer(20000);

// Load Landsat 8 collection for 2020
var collection = ee.ImageCollection('LANDSAT/LC08/C02/T1_TOA')
  .filterDate('2020-01-01', '2020-12-31')
  .filterBounds(roi);

// Add NDVI band to each image
var withNDVI = collection.map(function(image) {
  var ndvi = image.normalizedDifference(['B5', 'B4']).rename('NDVI');
  return image.addBands(ndvi);
});

// Different reducers for the collection
var mean = withNDVI.select('NDVI').mean();
var max = withNDVI.select('NDVI').max();
var median = withNDVI.select('NDVI').median();
var stdDev = withNDVI.select('NDVI').reduce(ee.Reducer.stdDev());

// Display the results
var vizParams = {min: 0, max: 1, palette: ['red', 'yellow', 'green']};
Map.centerObject(roi, 10);
Map.addLayer(mean, vizParams, 'Mean NDVI');
Map.addLayer(max, vizParams, 'Max NDVI');
Map.addLayer(median, vizParams, 'Median NDVI');
Map.addLayer(stdDev, {min: 0, max: 0.2, palette: ['blue', 'white', 'red']}, 'StdDev NDVI');

Time Series Analysis Project 
// Time Series Analysis Project

// Define a region of interest (e.g., a forest area)
var roi = ee.Geometry.Rectangle([-122.5, 37.5, -122.0, 38.0]);

// Load MODIS NDVI data for 10 years
var modisNDVI = ee.ImageCollection('MODIS/006/MOD13Q1')
  .filterDate('2010-01-01', '2020-12-31')
  .select('NDVI');

// Function to extract date from an image
function getDate(image) {
  return image.set('date', image.date().format('YYYY-MM-dd'));
}

// Map the function over the collection
var withDate = modisNDVI.map(getDate);

// Create a time series chart
var chart = ui.Chart.image.series({
  imageCollection: withDate,
  region: roi,
  reducer: ee.Reducer.mean(),
  scale: 250,
  xProperty: 'date'
}).setOptions({
  title: 'NDVI Time Series 2010-2020',
  vAxis: {title: 'NDVI', viewWindow: {min: 0, max: 10000}},
  hAxis: {title: 'Date', format: 'yyyy'},
  trendlines: {0: {
    color: 'red',
    lineWidth: 1,
    showR2: true,
    visibleInLegend: true
  }},
  lineWidth: 1,
  pointSize: 3
});

print(chart);

// Calculate annual averages
var years = ee.List.sequence(2010, 2020);

var annualNDVI = ee.ImageCollection.fromImages(
  years.map(function(year) {
    var annual = modisNDVI
      .filter(ee.Filter.calendarRange(year, year, 'year'))
      .mean();
    return annual
      .set('year', year)
      .set('system:time_start', ee.Date.fromYMD(year, 1, 1));
  })
);

// Create an annual chart
var annualChart = ui.Chart.image.series({
  imageCollection: annualNDVI,
  region: roi,
  reducer: ee.Reducer.mean(),
  scale: 250,
  xProperty: 'year'
}).setOptions({
  title: 'Annual Average NDVI 2010-2020',
  vAxis: {title: 'NDVI'},
  hAxis: {title: 'Year', format: '####'},
  lineWidth: 2,
  pointSize: 6
});

print(annualChart);

// Display the region on the map
Map.centerObject(roi, 10);
Map.addLayer(roi, {color: 'blue'}, 'Study Area');

// Display the most recent NDVI
var recent = modisNDVI.sort('system:time_start', false).first();
Map.addLayer(recent, {min: 0, max: 9000, palette: ['brown', 'yellow', 'green']}, 'Recent NDVI');
Copied!

Day 9: Project – Machine Learning (Land Use Land Cover - LULC)

  • Machine Learning in GEE: Supervised Classification (Random Forest, SVM, etc.)
  • Training and Testing Data Preparation
  • Creating LULC Maps using Classification Algorithms
Machine Learning for Land Use Land Cover Classification 
// Machine Learning for Land Use Land Cover Classification

// Define a region of interest
var roi = ee.Geometry.Rectangle([-122.5, 37.5, -122.0, 38.0]);

// Load Landsat 8 image
var landsat = ee.Image('LANDSAT/LC08/C02/T1_TOA/LC08_044034_20140318');

// Select the bands to use for classification
var bands = ['B2', 'B3', 'B4', 'B5', 'B6', 'B7'];

// Create training data for different land cover types
var urban = ee.FeatureCollection([
  ee.Feature(ee.Geometry.Point([-122.26, 37.87]), {class: 0}),
  ee.Feature(ee.Geometry.Point([-122.20, 37.75]), {class: 0}),
  ee.Feature(ee.Geometry.Point([-122.30, 37.79]), {class: 0})
]);

var water = ee.FeatureCollection([
  ee.Feature(ee.Geometry.Point([-122.35, 37.80]), {class: 1}),
  ee.Feature(ee.Geometry.Point([-122.45, 37.74]), {class: 1}),
  ee.Feature(ee.Geometry.Point([-122.25, 37.85]), {class: 1})
]);

var forest = ee.FeatureCollection([
  ee.Feature(ee.Geometry.Point([-122.14, 37.85]), {class: 2}),
  ee.Feature(ee.Geometry.Point([-122.10, 37.76]), {class: 2}),
  ee.Feature(ee.Geometry.Point([-122.08, 37.81]), {class: 2})
]);

// Merge the training data
var trainingData = urban.merge(water).merge(forest);

// Sample the input imagery to get training data
var training = landsat.select(bands).sampleRegions({
  collection: trainingData,
  properties: ['class'],
  scale: 30
});

// Train a Random Forest classifier
var classifier = ee.Classifier.smileRandomForest(50)
  .train({
    features: training,
    classProperty: 'class',
    inputProperties: bands
  });

// Classify the image
var classified = landsat.select(bands).classify(classifier);

// Define visualization parameters
var classViz = {
  min: 0,
  max: 2,
  palette: ['red', 'blue', 'green']
};

// Display the results
Map.centerObject(roi, 10);
Map.addLayer(landsat, {bands: ['B4', 'B3', 'B2'], min: 0, max: 0.3}, 'Landsat');
Map.addLayer(classified, classViz, 'Classification');

// Calculate area for each class
var areaImage = ee.Image.pixelArea().addBands(classified);
var areas = areaImage.reduceRegion({
  reducer: ee.Reducer.sum().group({
    groupField: 1,
    groupName: 'class',
  }),
  geometry: roi,
  scale: 30,
  maxPixels: 1e9
});

print('Class Areas (m²):', areas);

Advanced Topics - OSM Data Import 
// Advanced Topics - OSM Data Import

// Define a region of interest (e.g., San Francisco)
var roi = ee.Geometry.Rectangle([-122.5, 37.7, -122.3, 37.85]);

// Import OSM data using the OSM feature collection
var osmRoads = ee.FeatureCollection('OSM/SanFrancisco/Roads');
var osmBuildings = ee.FeatureCollection('OSM/SanFrancisco/Buildings');

// Filter roads by type
var mainRoads = osmRoads.filter(ee.Filter.or(
  ee.Filter.eq('highway', 'motorway'),
  ee.Filter.eq('highway', 'trunk'),
  ee.Filter.eq('highway', 'primary')
));

// Load Sentinel-2 imagery
var sentinel = ee.ImageCollection('COPERNICUS/S2_SR')
  .filterDate('2020-06-01', '2020-09-01')
  .filterBounds(roi)
  .filter(ee.Filter.lt('CLOUDY_PIXEL_PERCENTAGE', 20))
  .median();

// Display the data
Map.centerObject(roi, 12);
Map.addLayer(sentinel, {bands: ['B4', 'B3', 'B2'], min: 0, max: 3000}, 'Sentinel-2');
Map.addLayer(osmBuildings, {color: 'yellow'}, 'OSM Buildings');
Map.addLayer(mainRoads, {color: 'red'}, 'OSM Main Roads');

// Create a buffer around roads
var roadBuffer = mainRoads.map(function(feature) {
  return feature.buffer(100);
});

Map.addLayer(roadBuffer, {color: 'orange', opacity: 0.5}, 'Road Buffer');

// Calculate NDVI
var ndvi = sentinel.normalizedDifference(['B8', 'B4']).rename('NDVI');

// Calculate average NDVI near roads
var roadNDVI = ndvi.reduceRegions({
  collection: roadBuffer,
  reducer: ee.Reducer.mean(),
  scale: 10
});

print('NDVI near roads:', roadNDVI);

// Create a chart of NDVI distribution
var chart = ui.Chart.feature.histogram({
  features: roadNDVI,
  property: 'mean',
  maxBuckets: 30
}).setOptions({
  title: 'NDVI Distribution Near Roads',
  hAxis: {title: 'NDVI'},
  vAxis: {title: 'Count'}
});

print(chart);
