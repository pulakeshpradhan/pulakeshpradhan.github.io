Google Earth Engine Applications

Exploring Geospatial Analysis Across Disciplines

Google Earth Engine (GEE) is a cloud-based platform that enables users to visualize and analyze satellite images of our planet. It provides access to a multi-petabyte catalog of satellite imagery and geospatial datasets for planetary-scale analysis. This page showcases various applications of GEE across different disciplines, along with sample code and datasets.

Environmental scientists use Google Earth Engine to monitor and analyze changes in the Earth's environment, from deforestation and land use changes to climate impacts and natural disasters.

Topics Analyzed with Remote Sensing

  • Deforestation Analysis: Monitoring forest loss and degradation over time
  • Land Cover Change: Analyzing shifts in vegetation, urban areas, and natural landscapes
  • Air Quality Assessment: Measuring pollutants and their dispersion patterns

Remote Sensing Techniques

  • Multi-temporal Analysis: Comparing imagery over time to detect changes
  • Spectral Indices: Using band combinations to highlight environmental features

Sea Surface Temperature Analysis JavaScript 
// Sea Surface Temperature (SST) analysis using NOAA data
var startDate = '2020-01-01';
var endDate = '2020-12-31';
var region = ee.Geometry.Rectangle([-180, -50, 180, 50]);

// Load NOAA Optimum Interpolation SST dataset
var sst = ee.ImageCollection('NOAA/CDR/OISST/V2_1')
  .filterDate(startDate, endDate)
  .select('sst');

// Calculate monthly averages
var months = ee.List.sequence(1, 12);
var monthlySST = ee.ImageCollection.fromImages(
  months.map(function(month) {
    var filtered = sst.filter(ee.Filter.calendarRange(month, month, 'month'));
    return filtered.mean()
      .set('month', month)
      .set('system:time_start', ee.Date.fromYMD(2020, month, 1).millis());
  })
);

// Create an RGB visualization
var palette = ['navy', 'blue', 'cyan', 'lime', 'yellow', 'red'];
var visParams = {min: 0, max: 32, palette: palette};

// Display the annual mean SST
Map.addLayer(sst.mean(), visParams, 'Annual Mean SST');

// Create a chart of monthly SST for a point
var point = ee.Geometry.Point([0, 0]);
var chart = ui.Chart.image.series({
  imageCollection: monthlySST,
  region: point,
  reducer: ee.Reducer.mean(),
  scale: 27830
}).setOptions({
  title: 'Monthly Sea Surface Temperature',
  vAxis: {title: 'Temperature (°C)'},
  hAxis: {title: 'Month', format: 'M', gridlines: {count: 12}}

Coral Reef Bleaching Analysis JavaScript 
// Coral reef bleaching analysis using NOAA Coral Reef Watch data
var startDate = '2020-01-01';
var endDate = '2020-12-31';
var reefRegion = ee.Geometry.Rectangle([140, -25, 155, -10]); // Great Barrier Reef region

// Load NOAA Coral Reef Watch data
var coralReefWatch = ee.ImageCollection('NOAA/CDR/OISST/V2_1')
  .filterDate(startDate, endDate)
  .filterBounds(reefRegion)
  .select('sst');

// Calculate monthly maximum SST
var months = ee.List.sequence(1, 12);
var monthlyMaxSST = ee.ImageCollection.fromImages(
  months.map(function(month) {
    var filtered = coralReefWatch.filter(ee.Filter.calendarRange(month, month, 'month'));
    return filtered.max()
      .set('month', month)
      .set('system:time_start', ee.Date.fromYMD(2020, month, 1).millis());
  })
);

// Define bleaching threshold (28.5°C for Great Barrier Reef)
var bleachingThreshold = 28.5;

// Create a mask for areas exceeding the threshold
var bleachingRisk = monthlyMaxSST.map(function(image) {
  return image.gt(bleachingThreshold).rename('bleaching_risk')
    .addBands(image.subtract(bleachingThreshold).rename('temperature_excess'))
    .set('system:time_start', image.get('system:time_start'));
});

// Display the maximum SST for the year
var sstPalette = ['blue', 'cyan', 'lime', 'yellow', 'orange', 'red'];
Map.addLayer(coralReefWatch.max(), {min: 25, max: 32, palette: sstPalette}, 'Maximum SST');

// Display areas at risk of bleaching (maximum exposure)
Map.addLayer(bleachingRisk.select('bleaching_risk').max(), 
  {min: 0, max: 1, palette: ['white', 'red']}, 'Bleaching Risk Areas');

Sample GEE Code

// Land cover classification using Dynamic World dataset
var region = ee.Geometry.Rectangle([30, -5, 40, 5]); // East Africa
var year = 2022;

// Load Dynamic World data
var dynamicWorld = ee.ImageCollection('GOOGLE/DYNAMICWORLD/V1')
  .filterDate(year + '-01-01', year + '-12-31')
  .filterBounds(region);

// Get the probability bands
var probBands = ['water', 'trees', 'grass', 'flooded_vegetation', 
                 'crops', 'shrub_and_scrub', 'built', 'bare', 'snow_and_ice'];

// Create a composite of the probability bands
var dwComposite = dynamicWorld.select(probBands).mean();

// Get the most likely class
var classification = dwComposite.argmax().rename('classification');

// Define visualization parameters
var dwVisParams = {
  min: 0,
  max: 8,
  palette: [
    '#419BDF', // water
    '#397D49', // trees
    '#88B053', // grass
    '#7A87C6', // flooded_vegetation
    '#E49635', // crops
    '#DFC35A', // shrub_and_scrub
    '#C4281B', // built
    '#A59B8F', // bare
    '#B39FE1'  // snow_and_ice
  ]
};

// Display the classification
Map.centerObject(region, 6);
Map.addLayer(classification, dwVisParams, 'Land Cover Classification');

// Calculate area of cropland
var cropland = classification.eq(4); // 4 is the class index for crops
var croplandArea = cropland.multiply(ee.Image.pixelArea()).reduceRegion({
  reducer: ee.Reducer.sum(),
  geometry: region,
  scale: 100,
  maxPixels: 1e9
});

print('Cropland area (sq km):', ee.Number(croplandArea.get('classification')).divide(1e6));

Sample GEE Code

// Urban heat island analysis using Landsat 8 thermal data
var startDate = '2022-06-01';
var endDate = '2022-08-31';
var city = ee.Geometry.Rectangle([-74.3, 40.5, -73.7, 41.0]); // New York City

// Load Landsat 8 Collection 2 Tier 1 data
var landsat = ee.ImageCollection('LANDSAT/LC08/C02/T1_L2')
  .filterDate(startDate, endDate)
  .filterBounds(city)
  .filter(ee.Filter.lt('CLOUD_COVER', 20));

// Function to calculate Land Surface Temperature (LST)
var calculateLST = function(image) {
  // Get the thermal band (ST_B10 is surface temperature in Kelvin)
  var thermal = image.select('ST_B10');
  
  // Convert from Kelvin to Celsius
  var lstC = thermal.subtract(273.15).rename('LST');
  
  // Get NDVI for urban vs. vegetation analysis
  var ndvi = image.normalizedDifference(['SR_B5', 'SR_B4']).rename('NDVI');
  
  return image.addBands([lstC, ndvi]);
};

// Map the function over the collection
var lstCollection = landsat.map(calculateLST);

// Create a summer composite
var summerLST = lstCollection.select('LST').mean();
var summerNDVI = lstCollection.select('NDVI').mean();

// Define visualization parameters
var lstPalette = ['blue', 'cyan', 'green', 'yellow', 'orange', 'red'];
var lstParams = {min: 20, max: 40, palette: lstPalette};

// Display the results
Map.centerObject(city, 10);
Map.addLayer(summerLST, lstParams, 'Land Surface Temperature (°C)');

// Calculate urban-rural temperature difference
var urban = summerNDVI.lt(0.2); // Urban areas have low NDVI
var rural = summerNDVI.gt(0.5); // Rural/vegetated areas have high NDVI

var urbanTemp = summerLST.updateMask(urban).reduceRegion({
  reducer: ee.Reducer.mean(),
  geometry: city,
  scale: 100,
  maxPixels: 1e9
}).get('LST');

var ruralTemp = summerLST.updateMask(rural).reduceRegion({
  reducer: ee.Reducer.mean(),
  geometry: city,
  scale: 100,
  maxPixels: 1e9
}).get('LST');

// Calculate the urban heat island intensity
var uhiIntensity = ee.Number(urbanTemp).subtract(ruralTemp);
print('Urban Heat Island Intensity (°C):', uhiIntensity);

Urban Growth Modeling JavaScript 
// Urban growth modeling using cellular automata in GEE
var region = ee.Geometry.Rectangle([77.5, 12.8, 78.0, 13.2]); // Bangalore, India
var startYear = 2000;
var endYear = 2020;

// Load urban extent data for start and end years
var urbanStart = ee.Image('GOOGLE/DYNAMICWORLD/V1')
  .filterDate(startYear + '-01-01', startYear + '-12-31')
  .select('built').mean().gt(0.5);

var urbanEnd = ee.Image('GOOGLE/DYNAMICWORLD/V1')
  .filterDate(endYear + '-01-01', endYear + '-12-31')
  .select('built').mean().gt(0.5);

// Calculate urban growth rate
var urbanGrowth = urbanEnd.subtract(urbanStart);
var urbanGrowthArea = urbanGrowth.multiply(ee.Image.pixelArea()).reduceRegion({
  reducer: ee.Reducer.sum(),
  geometry: region,
  scale: 10,
  maxPixels: 1e9
});

// Create a suitability map for urban growth
var dem = ee.Image('USGS/SRTMGL1_003').clip(region);
var slope = ee.Terrain.slope(dem);
var roads = ee.FeatureCollection('projects/google/examples/tiger-2018-roads')
  .filterBounds(region);

// Distance to existing urban areas
var distToUrban = urbanStart.fastDistanceTransform().sqrt().multiply(10);

// Distance to roads
var distToRoads = roads.distance().clip(region);

// Create suitability map (higher values = more suitable)
var suitability = ee.Image(1.0)
  .subtract(slope.divide(90)) // Lower slopes are more suitable
  .multiply(distToUrban.lt(2000)) // Areas close to existing urban are suitable
  .multiply(distToRoads.lt(1000)) // Areas close to roads are suitable
  .updateMask(urbanStart.not()); // Mask out existing urban areas

// Display results
Map.centerObject(region, 11);
Map.addLayer(urbanStart, {palette: ['white', 'red']}, 'Urban ' + startYear);
Map.addLayer(urbanEnd, {palette: ['white', 'blue']}, 'Urban ' + endYear);
Map.addLayer(urbanGrowth, {palette: ['white', 'purple']}, 'Urban Growth');
Map.addLayer(suitability, {min: 0, max: 1, palette: ['white', 'yellow', 'orange', 'red']}, 'Growth Suitability');

Time Series Analysis with Landsat JavaScript 
// Time series analysis of vegetation trends using Landsat
var geometry = ee.Geometry.Point([-122.2, 37.4]); // San Francisco Bay Area
var startYear = 1985;
var endYear = 2023;

// Function to calculate NDVI for Landsat 5
function addNDVI_L5(image) {
  var ndvi = image.normalizedDifference(['B4', 'B3']).rename('NDVI');
  return image.addBands(ndvi);
}

// Function to calculate NDVI for Landsat 8
function addNDVI_L8(image) {
  var ndvi = image.normalizedDifference(['B5', 'B4']).rename('NDVI');
  return image.addBands(ndvi);
}

// Create a list of years
var years = ee.List.sequence(startYear, endYear);

// Function to calculate annual NDVI composite
var annualNDVI = ee.ImageCollection.fromImages(
  years.map(function(year) {
    // Get Landsat 5 data for 1985-2011
    var l5Collection = ee.ImageCollection('LANDSAT/LT05/C02/T1_L2')
      .filterDate(ee.Number(year).format('%d'), ee.Number(year).add(1).format('%d'))
      .filterBounds(geometry)
      .filter(ee.Filter.lt('CLOUD_COVER', 20))
      .map(addNDVI_L5);
    
    // Get Landsat 8 data for 2013-present
    var l8Collection = ee.ImageCollection('LANDSAT/LC08/C02/T1_L2')
      .filterDate(ee.Number(year).format('%d'), ee.Number(year).add(1).format('%d'))
      .filterBounds(geometry)
      .filter(ee.Filter.lt('CLOUD_COVER', 20))
      .map(addNDVI_L8);
    
    // Merge collections based on year
    var collection = ee.Algorithms.If(
      ee.Number(year).lt(2012),
      l5Collection,
      l8Collection
    );
    
    // Calculate annual maximum NDVI
    var annualMax = ee.ImageCollection(collection).select('NDVI').max()
      .set('year', year)
      .set('system:time_start', ee.Date.fromYMD(year, 1, 1).millis());
    
    return annualMax;
  })
);

// Create a chart of NDVI over time
var chart = ui.Chart.image.series({
  imageCollection: annualNDVI,
  region: geometry,
  reducer: ee.Reducer.mean(),
  scale: 30
}).setOptions({
  title: 'NDVI Time Series (1985-2023)',
  vAxis: {title: 'NDVI'},
  hAxis: {title: 'Year', format: 'yyyy'},
  trendlines: {0: {color: 'red'}}
});

print(chart);

// Display the slope of the trend (positive = greening, negative = browning)

Flood Mapping with SAR JavaScript 
// Flood mapping using Sentinel-1 SAR data
var aoi = ee.Geometry.Rectangle([90.2, 23.5, 91.0, 24.2]); // Bangladesh

// Define pre-flood and flood periods
var preFloodStart = '2020-05-01';
var preFloodEnd = '2020-05-15';
var floodStart = '2020-07-15';
var floodEnd = '2020-07-30';

// Load Sentinel-1 SAR GRD data
var collection = ee.ImageCollection('COPERNICUS/S1_GRD')
  .filter(ee.Filter.eq('instrumentMode', 'IW'))
  .filter(ee.Filter.listContains('transmitterReceiverPolarisation', 'VH'))
  .filter(ee.Filter.eq('orbitProperties_pass', 'DESCENDING'))
  .filterBounds(aoi)
  .select('VH');

// Get pre-flood and flood images
var preFloodCollection = collection.filterDate(preFloodStart, preFloodEnd);
var floodCollection = collection.filterDate(floodStart, floodEnd);

var preFloodImage = preFloodCollection.mean();
var floodImage = floodCollection.mean();

// Apply speckle filter
var smoothingRadius = 50;
var preFloodFiltered = preFloodImage.focal_mean(smoothingRadius, 'circle', 'meters');
var floodFiltered = floodImage.focal_mean(smoothingRadius, 'circle', 'meters');

// Calculate the difference
var difference = preFloodFiltered.subtract(floodFiltered);

// Threshold the difference to identify flooded areas
// Water appears darker in SAR imagery (lower backscatter)
var threshold = 2; // Adjust based on local conditions
var flooded = difference.gt(threshold);

// Apply a minimum mapping unit to remove noise
var connectedPixels = flooded.connectedPixelCount(100, true);
var floodRefined = flooded.updateMask(connectedPixels.gt(10));

// Get permanent water to exclude from flood map
var permanentWater = ee.Image('JRC/GSW1_4/GlobalSurfaceWater')
  .select('occurrence')
  .gt(90)
  .clip(aoi);

// Create final flood map (excluding permanent water)
var floodMap = floodRefined.updateMask(permanentWater.not());

// Calculate flood area
var floodArea = floodMap.multiply(ee.Image.pixelArea()).reduceRegion({
  reducer: ee.Reducer.sum(),
  geometry: aoi,
  scale: 30,
  maxPixels: 1e9
});

// Display results
Map.centerObject(aoi, 10);
Map.addLayer(preFloodFiltered, {min: -25, max: 0}, 'Pre-flood SAR');
Map.addLayer(floodFiltered, {min: -25, max: 0}, 'Flood SAR');
Map.addLayer(floodMap, {palette: ['blue']}, 'Flood Extent');

print('Flood area (sq km): ', ee.Number(floodArea.get('VH')).divide(1e6));
